import telebot
import unicodedata

TOKEN = 'YOUR_TOKEN_HERE'   # ← Replace with your token

bot = telebot.TeleBot(TOKEN)

def contains_non_english_alphabet(text):
    """Returns True if message contains any non-Latin (non-English) letters."""
    if not text:
        return False
    for char in text:
        if char.isalpha():                    # only check letters
            try:
                name = unicodedata.name(char)
                if 'LATIN' not in name:       # Cyrillic, Arabic, CJK, etc. have no "LATIN"
                    return True
            except ValueError:
                return True                   # unknown char → treat as non-English
    return False

@bot.message_handler(func=lambda m: True)
def check_message(message):
    # Only work in groups/supergroups
    if message.chat.type not in ['group', 'supergroup']:
        return

    # Check text OR caption (photos, videos, etc. with text)
    text = (message.text or message.caption or "").strip()
    if not text:
        return

    if contains_non_english_alphabet(text):
        try:
            # Delete the bad message
            bot.delete_message(message.chat.id, message.message_id)

            # Friendly warning with mention
            user = message.from_user
            name = user.first_name or "User"
            mention = f'<a href="tg://user?id={user.id}">{name}</a>'
            warning = f"⚠️ {mention}, **only English (Latin alphabet) is allowed** in this group!"
            
            bot.send_message(message.chat.id, warning, parse_mode='HTML')
        except Exception as e:
            print(f"Error: {e}")

# Optional: start message
@bot.message_handler(commands=['start'])
def start(message):
    bot.reply_to(message, "✅ English-only bot is active!\nNon-English alphabet messages will be deleted.")

print("🤖 Bot started... (press Ctrl+C to stop)")
bot.infinity_polling()
